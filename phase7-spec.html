// 问题: 23.976fps = 24000/1001 fps

// 当前简化: PTS = ms * 90 = frame_number * 1000 / 24 * 90 = frame_number * 3750

    if is_ntsc {
        // 精确计算: ms * 90000 * 1001 / 1000000
        (ms as u128 * 90_000 * 1001 / 1_000_000) as u64
    } else {
        ms * 90
    }
}

1.1 — 坐标变换系统 (Transform Pipeline)

★★★★ 仿射变换: 旋转 + 缩放 + 剪切

现状: rotation/scale_x/scale_y/shear_x/shear_y 写入 RenderContext 但 render_event() 从未使用。

目标: 在字形光栅化后，对整行文字做仿射变换。

方案:

// 在 render_event() 中，光栅化完成后：
if ctx.rotation != 0.0 || ctx.scale_x != 100.0 || ctx.scale_y != 100.0 || ctx.shear_x != 0.0 {
    let origin = (ctx.origin_x, ctx.origin_y);
    // 如果没有 \org，使用文字行中心作为旋转原点
    if origin.0 == 0.0 && origin.1 == 0.0 {
        origin = (text_center_x, text_baseline_y);
    }
    apply_affine_transform(pixmap, origin, ctx.rotation, ctx.scale_x/100.0, ctx.scale_y/100.0, ctx.shear_x);
}

fn apply_affine_transform(
    pixmap: &mut Pixmap,
    origin: (f32, f32),
    rotation_deg: f32,
    scale_x: f32,
    scale_y: f32,
    shear_x: f32,
) {
    let rad = rotation_deg.to_radians();
    let cos = rad.cos();
    let sin = rad.sin();
    // 逆变换矩阵: 从目标像素反查源像素
    // M = T(-origin) * Shear * Scale * Rotate * T(origin)
    let src = pixmap.data().to_vec(); // 复制原图
    let data = pixmap.data_mut();
    for dy in 0..h {
        for dx in 0..w {
            // 逆变换: dst → src
            let fx = (dx as f32 - origin.0);
            let fy = (dy as f32 - origin.1);
            let sx = (fx * cos + fy * sin) / scale_x + origin.0 - shear_x * fy;
            let sy = (-fx * sin + fy * cos) / scale_y + origin.1;
            // 双线性插值采样
            if let Some(rgba) = bilinear_sample(&src, w, h, sx, sy) {
                data[idx..idx+4].copy_from_slice(&rgba);
            }
        }
    }
}

1.2 — 时间动画系统 (Time-based Animation)

★★★★★ \t / \move / \fad / \fade 时间插值

现状: 这些标签都是"解析了，但渲染时忽略时间维度"。

核心难点: 当前 render_ass(ass, timestamp_ms) 是单帧渲染，不感知"当前帧在整个事件时间线中的位置"。

1.2.1 — \move(x1,y1,x2,y2,t1,t2) 位置插值

// 在 build_context 中:
OverrideTag::Move { x1, y1, x2, y2, t1, t2 } => {
    let progress = interpolate_time(timestamp_ms, event.start.as_ms() + t1, event.start.as_ms() + t2);
    let x = x1 + (x2 - x1) * progress;
    let y = y1 + (y2 - y1) * progress;
    ctx.x = x as f32 * scale_x;
    ctx.y = y as f32 * scale_y;
}

fn interpolate_time(current: u64, start: u64, end: u64) -> f64 {
    if current <= start { 0.0 }
    else if current >= end { 1.0 }
    else { (current - start) as f64 / (end - start) as f64 }
}

1.2.2 — \fad(dur_in, dur_out) 淡入淡出

// 在 render_event 开头:
if let Some((fade_in, fade_out)) = get_fad_params(event) {
    let ts_in_event = timestamp_ms - event.start.as_ms();
    let event_duration = event.end.as_ms() - event.start.as_ms();
    let alpha_factor = if ts_in_event < fade_in {
        ts_in_event as f32 / fade_in as f32
    } else if ts_in_event > event_duration - fade_out {
        (event_duration - ts_in_event) as f32 / fade_out as f32
    } else {
        1.0
    };
    // 应用到所有颜色的 alpha 通道
    ctx.primary_color[3] = (ctx.primary_color[3] as f32 * alpha_factor) as u8;
    // ... 其他颜色
}

1.2.3 — \fade(a1,a2,a3,t1,t2,t3,t4) 三段透明度

// 5 段时间线: [0, t1] → [t1, t2] → [t2, t3] → [t3, t4] → [t4, end]
// alpha: a1 → a2 (线性) → a2 (常数) → a3 (线性) → a3 (常数)
fn compute_fade_alpha(ts_in_event: u64, params: &FadeComplexParams) -> u8 {
    let t = ts_in_event;
    if t <= params.t1 { lerp(255, params.alpha_start, t, params.t1) }
    else if t <= params.t2 { params.alpha_start }
    else if t <= params.t3 { lerp(params.alpha_start, params.alpha_mid, t - params.t2, params.t3 - params.t2) }
    else if t <= params.t4 { lerp(params.alpha_mid, params.alpha_end, t - params.t3, params.t4 - params.t3) }
    else { params.alpha_end }
}

1.2.4 — \t(tag, duration, accel) 属性渐变

// 这是最复杂的: 需要在两个 RenderContext 之间插值
// 1. 构建"起始 context"（当前 event 的 override tags，不含 \t）
// 2. 构建"目标 context"（应用 \t 内的 tags）
// 3. 在 t1..t2 时间范围内，对 context 的数值字段做 pow(progress, accel) 插值
// 4. 字符串字段（font_name）在 progress > 0.5 时切换

fn interpolate_context(start: &RenderContext, end: &RenderContext, progress: f64, accel: f64) -> RenderContext {
    let t = progress.powf(accel);
    RenderContext {
        font_size: lerp_f32(start.font_size, end.font_size, t),
        primary_color: lerp_color(&start.primary_color, &end.primary_color, t),
        outline_width: lerp_f32(start.outline_width, end.outline_width, t),
        rotation: lerp_f32(start.rotation, end.rotation, t),
        scale_x: lerp_f32(start.scale_x, end.scale_x, t),
        scale_y: lerp_f32(start.scale_y, end.scale_y, t),
        blur: lerp_f32(start.blur, end.blur, t),
        // ... 其他数值字段
        font_name: if t > 0.5 { end.font_name.clone() } else { start.font_name.clone() },
        ..start.clone()
    }
}

2 PGS 播放兼容性增强

2.1 — 非整数帧率精确处理

★★★ 23.976 / 29.97 / 59.94 帧率

现状: frame_rate_code() 只返回粗略 code (0x10/0x20/...)，PTS 计算用 ms * 90 不考虑 drop-frame。

问题: 23.976fps 实际是 24000/1001 ≈ 23.976023976...。如果用 24.0 计算 PTS，长字幕会有时间漂移。

方案:

// 精确 PTS 计算
fn ms_to_pts_exact(ms: u64, fps: f64) -> u64 {
    // 使用 90kHz 时钟，精确计算
    // PTS = round(ms * 90000 / 1000) 对于整数帧率
    // 对于 NTSC: PTS = round(ms * 90000 * 1001 / 1000 / 1000)
    let numerator = 90_000.0 * 1001.0;
    let denominator = 1000.0 * 1000.0;
    (ms as f64 * numerator / denominator).round() as u64
}

// 帧率 code 需要区分 NTSC
fn frame_rate_code_exact(fps: f64) -> (u8, bool) {
    // 返回 (code, is_ntsc)
    if (fps - 23.976).abs() < 0.01 { (0x10, true) }
    else if (fps - 29.97).abs() < 0.01 { (0x40, true) }
    else if (fps - 59.94).abs() < 0.01 { (0x70, true) }
    else { (frame_rate_code(fps), false) }
}

2.3 — PGS Decoder Buffer 限制

★★★★★ "大金过水" — ODS 分块 + Epoch 管理

问题: 蓝光播放器的 PGS decoder buffer 约 2MB。如果单个 Display Set (PCS+WDS+PDS+ODS+END) 总大小超过此限制，播放器会崩溃或跳过字幕。

典型场景: 大面积字幕（全屏歌词、复杂特效字幕）的 RLE 数据可能很大。

方案:

// 1. 估算 Display Set 总大小
fn estimate_display_set_size(frame: &QuantizedFrame) -> usize {
    let rle = rle_encode(&frame.indices, frame.width, frame.height);
    13 + pcs_size + 13 + wds_size + 13 + pds_size + 13 + rle.len() + 4 + 13 // headers + payloads
}

// 2. 如果超过限制，拆分为多个 Epoch
// Epoch 1: 清除旧字幕 (empty PCS)
// Epoch 2: 显示新字幕 (如果有空间)
// 或者: 降低分辨率/颜色数来减小 RLE 大小

fn split_epoch_if_needed(frame: &QuantizedFrame, max_buffer: usize) -> Vec<EpochChunk> {
    let total = estimate_display_set_size(frame);
    if total <= max_buffer {
        return vec![EpochChunk::Full(frame.clone())];
    }
    // 策略 1: 按行拆分
    // 策略 2: 降低量化颜色数 (256 → 128 → 64) 减小 RLE
    // 策略 3: 降低分辨率
    split_by_rows(frame, max_buffer)
}

3.1 — Fontconfig 风格匹配

★★★ Weight/Width/Style 精确匹配

现状: FontManager::query(family, bold, italic) 只做 family + bold(≥700) + italic 三路匹配。

问题: ASS 可以指定 \b (bold=700) 或 \b900 (black)。当前只区分 ≥700 和 <700。

方案:

// 增强 query 支持 weight 值
pub fn query_with_weight(&self, family: &str, weight: u16, italic: bool, width: u16) -> Option<ID> {
    let candidates: Vec<_> = self.db.faces()
        .filter(|f| f.families.iter().any(|(name, _)| name.eq_ignore_ascii_case(family)))
        .collect();

    // 评分: weight 差值 + italic 匹配 + width 差值
    candidates.iter()
        .min_by_key(|f| {
            let weight_diff = (f.weight.0 as i32 - weight as i32).unsigned_abs();
            let italic_penalty = if f.italic != italic { 100 } else { 0 };
            let width_diff = (f.stretch.0 as i32 - width as i32).unsigned_abs();
            weight_diff + italic_penalty + width_diff
        })
        .map(|f| f.id)
}
