```rust
fn is_png_blank(path: &std::path::Path) -> bool {
    let decoder = png::Decoder::new(std::fs::File::open(path).unwrap());
    let mut reader = decoder.read_info().unwrap();
    let mut buf = vec![0u8; reader.output_buffer_size()];
    let info = reader.next_frame(&mut buf).unwrap();
    let data = &buf[..info.buffer_size];
    let stride = info.width as usize * info.bytes_per_pixel;
    for y in 0..info.height as usize {
        let row = &data[y * stride..(y + 1) * stride];
        for pixel in row.chunks(info.bytes_per_pixel) {
            let alpha = pixel[3];
            let non_transparent = pixel[0] != 0 || pixel[1] != 0 || pixel[2] != 0;
            if alpha > 0 && non_transparent {
                let r = pixel[0] as f64;
                let g = pixel[1] as f64;
                let b = pixel[2] as f64;
                let brightness = (0.299 * r + 0.587 * g + 0.114 * b) / 255.0;
                if brightness > 0.01 && brightness < 0.99 {
                    return false;
                }
            }
        }
    }
    true
}
```

```rust
fn read_be16(data: &[u8], offset: usize) -> Option<u16> {
```
